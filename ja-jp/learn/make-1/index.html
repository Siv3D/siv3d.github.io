			.draw(Arg::top = ColorF{ 0.1, 0.4, 0.8 }, Arg::bottom = ColorF{ 0.3, 0.7, 1.0 });

		// „Éó„É¨„Ç§„É§„Éº„ÅÆ„ÉÜ„ÇØ„Çπ„ÉÅ„É£„ÅÆÊèèÁîª
		playerTexture.drawAt(playerPos);

		// „Ç¢„Ç§„ÉÜ„É†„ÅÆÊèèÁîª
		for (const auto& item : items)
		{
			itemTexture.drawAt(item.pos);
		}

		// ÂΩì„Åü„ÇäÂà§ÂÆö„ÅÆÂèØË¶ñÂåñÔºà„Éá„Éê„ÉÉ„Ç∞Áî®Ôºâ
		Circle{ playerPos, 60 }.drawFrame(2, Palette::Red); // „Éó„É¨„Ç§„É§„Éº„ÅÆÂΩì„Åü„ÇäÂà§ÂÆöÂÜÜ
		for (const auto& item : items)
		{
			// „Ç¢„Ç§„ÉÜ„É†„ÅÆÂΩì„Åü„ÇäÂà§ÂÆöÂÜÜ
			Circle{ item.pos, 60 }.drawFrame(2, Palette::Red);
		}
	}
