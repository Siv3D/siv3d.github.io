    // X „Éû„Éº„ÇØ„ÅÆÂÄ§
    static constexpr int32 X_Mark = 2;

    void update()
    {
        if (m_gameOver)
        {
            return;
        }

        // 3x3 „ÅÆ„Çª„É´
        for (auto p : step(Size(3, 3)))
        {
            // „Çª„É´
            const Rect cell(p * CellSize, CellSize);

            // „Çª„É´„ÅÆ„Éû„Éº„ÇØ
            const int32 mark = m_grid[p];

            // „Çª„É´„ÅåÁ©∫ÁôΩ„Åß„ÄÅ„Å™„Åä„Åã„Å§„ÇØ„É™„ÉÉ„ÇØ„Åï„Çå„Åü„Çâ
            if ((mark == 0) && cell.leftClicked())
            {
                // „Çª„É´„Å´ÁèæÂú®„ÅÆ„Éû„Éº„ÇØ„ÇíÊõ∏„ÅçËæº„ÇÄ
                m_grid[p] = m_currentMark;

                // ÁèæÂú®„ÅÆ„Éû„Éº„ÇØ„ÇíÂÖ•„ÇåÊõø„Åà„Çã
                m_currentMark = ((m_currentMark == O_Mark) ? X_Mark : O_Mark);

                // „Å§„Å™„Åå„Å£„Åü„É©„Ç§„É≥„ÇíÊé¢„Åô
                m_lines = CheckLines(m_grid);

                // Á©∫ÁôΩ„Çª„É´„Åå 0 „Å´„Å™„Çã„Åã„ÄÅ„Å§„Å™„Åå„Å£„Åü„É©„Ç§„É≥„ÅåË¶ã„Å§„Åã„Å£„Åü„Çâ
                if (m_grid.count(0) == 0 || m_lines)
                {
                    // „Ç≤„Éº„É†ÁµÇ‰∫Ü
                    m_gameOver = true;
                }
            }
        }
    }

    // „Ç≤„Éº„É†„Çí„É™„Çª„ÉÉ„Éà
    void reset()
    {
        m_currentMark = O_Mark;

        m_grid.fill(0);

        m_lines.clear();

        m_gameOver = false;
    }

    // ÊèèÁîª
    void draw() const
    {
        drawGridLines();

        drawCells();

        drawResults();
    }

    // „Ç≤„Éº„É†„ÅåÁµÇ‰∫Ü„Åó„Åü„Åã„ÇíËøî„Åô
    bool isGameOver() const
    {
        return m_gameOver;
    }
};

void Main()
{
    // ËÉåÊôØËâ≤
    Scene::SetBackground(ColorF(0.8, 1.0, 0.9));

    constexpr Point offset(175, 30);

    GameBoard gameBoard;

    while (System::Update())
    {
        {
            // 2D ÊèèÁîª„Å®„Éû„Ç¶„Çπ„Ç´„Éº„ÇΩ„É´Â∫ßÊ®ô„ÇíÁßªÂãï
            Transformer2D tr(Mat3x2::Translate(offset), true);

            gameBoard.update();

            gameBoard.draw();
        }

        // „Ç≤„Éº„É†„ÅåÁµÇ‰∫Ü„Åó„Å¶„ÅÑ„Åü„Çâ
        if (gameBoard.isGameOver())
        {
            // Reset „Éú„Çø„É≥„ÇíÊäº„Åõ„Å∞„É™„Çª„ÉÉ„Éà
            if (SimpleGUI::ButtonAt(U"Reset", Vec2(400, 520)))
            {
                gameBoard.reset();
            }
        }
    }
}
