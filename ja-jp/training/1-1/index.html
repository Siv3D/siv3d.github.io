    // ÂÜÜ„ÇíÊèèÁîª„Åô„Çã„É°„É≥„ÉêÈñ¢Êï∞
    void draw() const
    {
        Circle(center, CircleR).draw(color);
    }
};

class CircleItemManager
{
private:

    // CircleItem „ÇíÊ†ºÁ¥ç„Åô„ÇãÈÖçÂàó
    Array<CircleItem> m_circleItems;

    // „Ç®„Éï„Çß„ÇØ„Éà
    Effect m_effect;

public:

    // „Éá„Éï„Ç©„É´„Éà„Ç≥„É≥„Çπ„Éà„É©„ÇØ„Çø
    CircleItemManager() = default;

    // Êñ∞„Åó„ÅÑ CircleItem „ÇíËøΩÂä†„Åô„Çã„É°„É≥„ÉêÈñ¢Êï∞
    void add(const Vec2& center, const ColorF& color)
    {
        // ÈÖçÂàó„Å´ CircleItem „ÇíËøΩÂä†
        m_circleItems.emplace_back(center, color);

        // „Ç®„Éï„Çß„ÇØ„Éà„ÇíËøΩÂä†
        m_effect.add([center](double t)
        {
            const double e = EaseOutExpo(t);
            Circle(center, CircleItem::CircleR + e * 30)
                .drawFrame(0, 4 * e, ColorF(1.0, (1 - e)));
            return t < 1.0;
        });
    }

    // „Åô„Åπ„Å¶„ÅÆ CircleItem „Å®„Ç®„Éï„Çß„ÇØ„Éà„ÇíÊèèÁîª„Åô„Çã„É°„É≥„ÉêÈñ¢Êï∞
    void draw() const
    {
        // „Ç®„Éï„Çß„ÇØ„Éà„ÇíÊõ¥Êñ∞„ÉªÊèèÁîª
        m_effect.update();

        // ÈÖçÂàóÂÜÖ„ÅÆ CircleItem „ÇíÊèèÁîª
        for (const auto& circleItem : m_circleItems)
        {
            circleItem.draw();
        }
    }
};

void Main()
{
    CircleItemManager circleItemManager;

    while (System::Update())
    {
        // „ÇÇ„Åó„Ç∑„Éº„É≥„ÅåÂ∑¶„ÇØ„É™„ÉÉ„ÇØ„Åï„Çå„Åü„Çâ
        if (Scene::Rect().leftClicked())
        {
            // ËøΩÂä†
            circleItemManager.add(Cursor::Pos(), RandomColorF());
        }

        // ÊèèÁîª
        circleItemManager.draw();
    }
}
